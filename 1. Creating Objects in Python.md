
Let's create a `Contact` class and one instance for a contact named Alice. 

```python
class Contact:
    def __init__(self, id, first_name, last_name, email):
        self.id = id
        self.first_name = first_name
        self.last_name = last_name
        self.email = email

# Example usage  
if __name__ == "__main__":  
	contact1 = Contact(1, "Alice", "Smith", "alice@example.com")  
	print(f"Contact: {contact1.first_name} {contact1.last_name}, Email: {contact1.email}")
```

Let's create another instance for another one named bobby 

```python
contact2 = Contact(2, "Bobby", "Raymond", "bob@example.com")  
print(f"Contact: {contact2.first_name} {contact2.last_name}, Email: {contact2.email}")
```


We can actually package both of these contacts into a list like this, and then loop through them. 

```python 
class Contact:  
    def __init__(self, id, first_name, last_name, email):  
        self.id = id  
        self.first_name = first_name  
        self.last_name = last_name  
        self.email = email  

if __name__ == "__main__":  
	contacts = [  
	    Contact(1,"Alice","Smith","alice@example.com"),  
	    Contact(2,"Bobby","Raymond","bob@example.com")  
	]  
	  
	for c in contacts:  
	    print(f"Contact: {c.first_name} {c.last_name}, Email: {c.email}")
```

Another pattern you might see is storing objects in a dictionary, and using one of the values as a key. 

```python
contacts_dict = {
    1: Contact(1, "Alice", "Smith", "alice@example.com"),
    2: Contact(2, "Bob", "Jones", "bob@example.com")
}

contact = contacts_dict[1]
print(f"Contact ID 1: {contact.first_name} {contact.last_name}")
```

Let's see how we can write objects to a CSV, comma-separated values. 

```python
import csv  
  
class Contact:  
    def __init__(self, id, first_name, last_name, email):  
        self.id = id  
        self.first_name = first_name  
        self.last_name = last_name  
        self.email = email  

if __name__ == "__main__":  
	# write contacts to CSV file  
	contacts_to_write = [  
	    Contact(1, "Alice", "Smith", "alice@example.com"),  
	    Contact(2, "Bob", "Jones", "bob@example.com"),  
	    Contact(3, "Charlie", "Brown", "charlie@example.com"),  
	    Contact(4, "Dana", "White", "dana@example.com")  
	]  
	  
	with open("contacts.csv", "w", newline="") as file:  
	    writer = csv.writer(file)  
	    writer.writerow(["id", "first_name", "last_name", "email"])  # Write header  
	    for contact in contacts_to_write:  
	        writer.writerow([contact.id, contact.first_name, contact.last_name, contact.email])  
	  
	# Read contacts from CSV and make objects  
	contacts_from_csv = []  
	with open("contacts.csv", "r", newline="") as file:  
	    reader = csv.reader(file)  
	    next(reader)  # Skip header row  
	    for row in reader:  
	        # Convert id to int, keep other fields as strings  
	        contact = Contact(int(row[0]), row[1], row[2], row[3])  
	        contacts_from_csv.append(contact)  
	  
	# Print contacts from CSV  
	for contact in contacts_from_csv:  
	    print(f"ID: {contact.id}, Name: {contact.first_name} {contact.last_name}, Email: {contact.email}")
```

import sqlite3  
  
class Contact:  
    def __init__(self, id, first_name, last_name, email):  
        self.id = id  
        self.first_name = first_name  
        self.last_name = last_name  
        self.email = email  
  
# Create/connect to SQLite database  
conn = sqlite3.connect("contacts.db")  
cursor = conn.cursor()  
  
# Create contacts table  
cursor.execute("""  
    CREATE TABLE IF NOT EXISTS contacts (        id INTEGER PRIMARY KEY,        first_name TEXT,        last_name TEXT,        email TEXT    )""")  
  
# Write contacts to SQLite  
contacts_to_write = [  
    Contact(1, "Alice", "Smith", "alice@example.com"),  
    Contact(2, "Bob", "Jones", "bob@example.com"),  
    Contact(3, "Charlie", "Brown", "charlie@example.com"),  
    Contact(4, "Dana", "White", "dana@example.com")  
]  
  
# Insert contacts into the table  
for contact in contacts_to_write:  
    cursor.execute(  
        "INSERT OR REPLACE INTO contacts (id, first_name, last_name, email) VALUES (?, ?, ?, ?)",  
        (contact.id, contact.first_name, contact.last_name, contact.email)  
    )  
conn.commit()  
  
# Read contacts from SQLite and make objects  
contacts_from_db = []  
cursor.execute("SELECT id, first_name, last_name, email FROM contacts")  
rows = cursor.fetchall()  
for row in rows:  
    # Create Contact instance from row data  
    contact = Contact(int(row[0]), row[1], row[2], row[3])  
    contacts_from_db.append(contact)  
  
# Close database connection  
conn.close()  
  
# Print contacts from database  
for contact in contacts_from_db:  
    print(f"ID: {contact.id}, Name: {contact.first_name} {contact.last_name}, Email: {contact.email}")
