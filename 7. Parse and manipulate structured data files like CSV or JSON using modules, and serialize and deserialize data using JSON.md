```python
import csv  
  
class Contact:  
    def __init__(self, first_name, last_name, email):  
        self.first_name = first_name  
        self.last_name = last_name  
        self.email = email  
  
contacts = [  
    Contact("John", "Doe", "john.doe@example.com"),  
    Contact("Jane", "Smith", "jane.smith@example.com"),  
    Contact("Bob", "Johnson", "bob.johnson@example.com"),  
    Contact("Alice", "Brown", "alice.brown@example.com"),  
    Contact("Charlie", "Davis", "charlie.davis@example.com"),  
    Contact("Emma", "Wilson", "emma.wilson@example.com"),  
    Contact("David", "Taylor", "david.taylor@example.com"),  
    Contact("Sarah", "Moore", "sarah.moore@example.com"),  
    Contact("Michael", "Lee", "michael.lee@example.com"),  
    Contact("Laura", "Clark", "laura.clark@example.com")  
]  
  
with open('contacts.csv', 'w', encoding='utf-8', newline='') as f:  
    writer = csv.writer(f)  
    writer.writerow(['first_name', 'last_name', 'email'])  
    for c in contacts:  
        writer.writerow([c.first_name, c.last_name, c.email])  
  
# now let's import the contacts  
contacts = []  
with open('contacts.csv', 'r', encoding='utf-8',newline='') as f:  
    reader = csv.DictReader(f)  
    for row in reader:  
        contacts.append(Contact(row['first_name'], row['last_name'], row['email']))  
  
# Print to verify  
for c in contacts:  
    print(f"{c.first_name} {c.last_name}: {c.email}")
```

```python
import json  
  
class Contact:  
    def __init__(self, first_name, last_name, email):  
        self.first_name = first_name  
        self.last_name = last_name  
        self.email = email  
  
contacts = [  
    Contact("John", "Doe", "john.doe@example.com"),  
    Contact("Jane", "Smith", "jane.smith@example.com"),  
    Contact("Bob", "Johnson", "bob.johnson@example.com"),  
    Contact("Alice", "Brown", "alice.brown@example.com"),  
    Contact("Charlie", "Davis", "charlie.davis@example.com"),  
    Contact("Emma", "Wilson", "emma.wilson@example.com"),  
    Contact("David", "Taylor", "david.taylor@example.com"),  
    Contact("Sarah", "Moore", "sarah.moore@example.com"),  
    Contact("Michael", "Lee", "michael.lee@example.com"),  
    Contact("Laura", "Clark", "laura.clark@example.com")  
]  
  
# Save contacts to JSON file  
with open('contacts.json', 'w', encoding='utf-8') as f:  
    json.dump([{  
        'first_name': c.first_name,  
        'last_name': c.last_name,  
        'email': c.email  
    } for c in contacts], f, indent=4)  
  
# Load contacts from JSON file  
contacts = []  
with open('contacts.json', 'r', encoding='utf-8') as f:  
    data = json.load(f)  
    for item in data:  
        contacts.append(Contact(item['first_name'], item['last_name'], item['email']))  
  
# Print to verify  
for c in contacts:  
    print(f"{c.first_name} {c.last_name}: {c.email}")
```

Let's take a look at another example. Let's take a JSON file that scraped all the movie data off Wikipedia, and print each record. 

```python
import requests  
import json  
  
url = "https://raw.githubusercontent.com/prust/wikipedia-movie-data/master/movies.json"  
response = requests.get(url)  
movies = json.loads(response.text)  
  
# loop through the movies  
for m in movies:  
    print(m)
```

```python
import csv
```
